Ниже приведена программа, записанная на пяти
языках программирования.



Бейсик

DIM s, t AS INTEGER
 INPUT s
 INPUT t
 IF s > 4 AND t > 4 THEN
    PRINT ‘YES’
 ELSE
    PRINT ‘NO’
 ENDIF

Python

s = int(input())
t = int(input())
if s > 4 and t > 4:
    print("YES")
else:
    print("NO")

Паскаль

var s, t: integer;
begin
    readln(s);
    readln(t);
    if (s > 4) and (t > 4)
        then writeln('YES')
        else writeln('NO')
end.

С++

#include <iostream>
using namespace std;
int main() {
    int s, t;
    cin >> s;
    cin >> t;
    if (s > 4 && t > 4)
        cout << "YES";
    else
        cout << "NO";
return 0;
}
