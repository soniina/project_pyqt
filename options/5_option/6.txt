 INPUT s
 INPUT t
 IF s > 12 OR t > 12 THEN
    PRINT ‘YES’
 ELSE
    PRINT ‘NO’
 ENDIF

using namespace std;
int main() {
    int s, t;
    cin >> s;
    cin >> t;

    else
        cout << "NO";
